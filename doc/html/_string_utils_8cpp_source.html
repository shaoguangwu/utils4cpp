56 UTILS4CPP_EXPORT std::string removeCharacter(const std::string& str, char ch, int option)
57 {
58  if (str.empty()) {
59  return str;
60  }
61 
62  if (option != 0) {
63  auto pos = option < 0 ? str.find(ch) : str.rfind(ch);
64  if (pos != std::string::npos) {
65  std::string result(str);
66  result.erase(pos, 1);
67  return result;
68  }
69  return str;
70  } else { // option == 0
71  auto pos = str.find(ch);
72  if (pos == std::string::npos) {
73  return str;
74  }
75 
76  std::unique_ptr<char[]> buff(new char[str.size() + 1]);
77  std::size_t len = 0;
78  std::string::size_type pos1 = 0;
79  while (pos != std::string::npos) {
80  std::copy_n(str.begin() + pos1, pos - pos1, buff.get() + len);
81  len += pos - pos1;
82  pos1 = ++pos;
83  pos = str.find(ch, pos1);
84  }
85  std::copy_n(str.begin() + pos1, str.size() - pos1, buff.get() + len);
86  len += str.size() - pos1;
87  return std::string(buff.get(), len);
88  }
89 }

262 {
263  std::vector<std::string> result;
264  std::size_t beg = 0;
265  std::size_t pos = str.find(delim);
266  while (pos != std::string::npos)
267  {
268  result.emplace_back(str.substr(beg, pos - beg));
269  beg = ++pos;
270  pos = str.find(delim, beg);
271  }
272  if (beg > 0) {
273  result.emplace_back(str.substr(beg));
274  }

518 UTILS4CPP_EXPORT std::string capitalizedWords(const std::string& str)
519 {
520  std::string result(str);
521 
522  for (std::size_t i = 0; i < str.size(); ++i) {
523 #if defined(_MSC_VER) && defined(_MT) && defined(_DEBUG)
524  // MS has an assert that will fail if s[i] < 0;
525  // setting LC_CTYPE using setlocale() does *not* help.
526  if ((int)str[i] >= 0 && std::isalpha((int)str[i]) &&
527  (i == 0 || ((int)str[i - 1] >= 0 && std::isspace((int)str[i - 1]))))
528 #else
529  if (std::isalpha((int)str[i]) && (i == 0 || std::isspace((int)str[i - 1])))
530 #endif
531  {
532  result[i] = static_cast<std::string::value_type>(std::toupper((int)str[i]));
533  }
534  }
535  return result;
536 }
