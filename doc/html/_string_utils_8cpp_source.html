81 UTILS4CPP_EXPORT std::vector<std::string> splitString(const std::string& str, char delim)
82 {
83  std::vector<std::string> result;
84  std::size_t beg = 0;
85  std::size_t pos = str.find(delim);
86  while (pos != std::string::npos)
87  {
88  result.push_back(str.substr(beg, pos - beg));
89  beg = ++pos;
90  pos = str.find(delim, beg);
91  }
92  if (beg > 0) {
93  result.push_back(str.substr(beg));
94  }
95 
96  return result;

338 UTILS4CPP_EXPORT std::string capitalizedWords(const std::string& str)
339 {
340  std::string result(str);
341 
342  for (std::size_t i = 0; i < str.size(); ++i) {
343 #if defined(_MSC_VER) && defined(_MT) && defined(_DEBUG)
344  // MS has an assert that will fail if s[i] < 0;
345  // setting LC_CTYPE using setlocale() does *not* help.
346  if ((int)str[i] >= 0 && std::isalpha((int)str[i]) &&
347  (i == 0 || ((int)str[i - 1] >= 0 && std::isspace((int)str[i - 1]))))
348 #else
349  if (std::isalpha((int)str[i]) && (i == 0 || std::isspace((int)str[i - 1])))
350 #endif
351  {
352  result[i] = static_cast<std::string::value_type>(std::toupper((int)str[i]));
353  }
354  }
355  return result;
356 }
