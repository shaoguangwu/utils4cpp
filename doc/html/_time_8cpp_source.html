126 bool Time::setTime(int h, int m, int s, int ms, int dst)
127 {
128  m_isdst = dst;

130  m_msecs = NullTime; // make this invalid
131  return false;
132  }

150 Time Time::addMSecs(milliseconds_t ms) const
151 {
152  Time result;
153  if (isValid()) {
154  if (ms < 0) {
155  // %,/ not well-defined for -ve, so always work with +ve.
156  int negdays = (MSECS_PER_DAY - ms) / MSECS_PER_DAY;
157  result.m_msecs = (toMilliseconds() + ms + negdays * MSECS_PER_DAY) % MSECS_PER_DAY;
158  } else {
159  result.m_msecs = (toMilliseconds() + ms) % MSECS_PER_DAY;
160  }
161  }
162  return result;
163 }
164 int Time::msecsTo(const Time& t) const
