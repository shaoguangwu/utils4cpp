205 template<typename ... Args>
206 std::string formatString(const std::string& format, Args ... args)
207 {
208  auto size = std::snprintf(nullptr, 0, format.c_str(), args ...) + 1; // Extra space for '\0'
209  if (size <= 0) {
210  return std::string();
211  }
212  std::unique_ptr<char[]> buf(new char[size]);
213  std::snprintf(buf.get(), size, format.c_str(), args ...);
214  return { buf.get(), buf.get() + size - 1 }; // We don't want the '\0' inside
215 }
216 
223 template<typename ... Args>
224 std::wstring formatString(const std::wstring& format, Args ... args)
225 {
226  auto size = std::swprintf(nullptr, 0, format.c_str(), args ...) + 1; // Extra space for '\0'
227  if (size <= 0) {
228  return std::wstring();
229  }
230  std::unique_ptr<wchar_t[]> buf(new wchar_t[size]);
231  std::swprintf(buf.get(), size, format.c_str(), args ...);
232  return { buf.get(), buf.get() + size - 1 }; // We don't want the '\0' inside
233 }
