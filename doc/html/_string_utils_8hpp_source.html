174 template<class StringT>
175 void removeCharacterSelf(StringT& str, typename StringT::value_type ch, int option)
176 {
177  using CharT = typename StringT::value_type;
178  using SizeT = typename StringT::size_type;
179 
180  if (str.empty()) {
181  return;
182  }
183 
184  if (option != 0) {
185  auto pos = option < 0 ? str.find(ch) : str.rfind(ch);
186  if (pos != StringT::npos) {
187  str.erase(pos, 1);
188  return;
189  }
190  }
191  else { // option == 0
192  auto pos = str.find(ch);
193  if (pos != StringT::npos) {
194  std::unique_ptr<CharT[]> buff(new CharT[str.size() + 1]);
195  SizeT len = 0;
196  SizeT pos1 = 0;
197  while (pos != StringT::npos) {
198  std::copy_n(str.begin() + pos1, pos - pos1, buff.get() + len);
199  len += pos - pos1;
200  pos1 = ++pos;
201  pos = str.find(ch, pos1);
202  }
203  std::copy_n(str.begin() + pos1, str.size() - pos1, buff.get() + len);
204  len += str.size() - pos1;
205  str = StringT(buff.get(), len);
206  }
207  }
208 }

298 {

309  return true;

313  }
314  return false;

638 template<>
639 std::string convertString(const std::wstring& str, const char* locale)
640 {
641  std::string ctype_locale;
642  auto prev_ctype_loc = std::setlocale(LC_CTYPE, locale);
643  if (prev_ctype_loc) {
644  ctype_locale = prev_ctype_loc;
645  }
646 
647  std::string result;
648  std::mbstate_t state = std::mbstate_t();
649  const wchar_t* wstr = str.c_str();
650 
651 #ifdef _MSC_VER
652 # pragma warning(push)
653 # pragma warning(disable:4996)
654 #endif // _MSC_VER
655  auto len = std::wcsrtombs(nullptr, &wstr, 0, &state) + 1;
656  if (len > 1) {
657  std::unique_ptr<char[]> buf(new char[len]);
658  std::wcsrtombs(buf.get(), &wstr, len, &state);
659  result = std::string(buf.get(), len - 1);
660  }
661 
662 #ifdef _MSC_VER
663 # pragma warning(pop)
664 #endif // _MSC_VER
665 
666  std::setlocale(LC_CTYPE, ctype_locale.c_str()); // restore locale-ctype
667  return result;
668 }
