195 template<typename ... Args>
196 std::string formatString(const std::string& format, Args ... args)
197 {
198  auto size = std::snprintf(nullptr, 0, format.c_str(), args ...) + 1; // Extra space for '\0'
199  std::unique_ptr<char[]> buf(new char[size]);
200  std::snprintf(buf.get(), size, format.c_str(), args ...);
201  return { buf.get(), buf.get() + size - 1 }; // We don't want the '\0' inside
202 }

210 template<typename ... Args>
