127 bool UTime::setTime(int h, int m, int s, int ms, int dst)
128 {
129  m_isdst = dst;
130  if (!isValid(h, m, s, ms)) {
131  m_msecs = NullTime; // make this invalid
132  return false;
133  }
134  m_msecs = (h * SECS_PER_HOUR + m * SECS_PER_MIN + s) * 1000 + ms;
135  return true;
136 }

143 int UTime::secsTo(const UTime& t) const
144 {
145  if (!isValid() || !t.isValid())
146  return 0;
147 
148  // Truncate milliseconds as we do not want to consider them.
149  return (t.toMilliseconds() / 1000) - (toMilliseconds() / 1000);

151 UTime UTime::addMSecs(milliseconds_t ms) const
152 {
153  UTime result;
154  if (isValid()) {
155  if (ms < 0) {
156  // %,/ not well-defined for -ve, so always work with +ve.
157  int negdays = (MSECS_PER_DAY - ms) / MSECS_PER_DAY;
158  result.m_msecs = (toMilliseconds() + ms + negdays * MSECS_PER_DAY) % MSECS_PER_DAY;
159  } else {
160  result.m_msecs = (toMilliseconds() + ms) % MSECS_PER_DAY;
161  }
162  }
163  return result;
164 }
165 int UTime::msecsTo(const UTime& t) const
166 {

168  return 0;
169  return t.toMilliseconds() - toMilliseconds();
