51 template<class CharT, class Traits, class SizeT>
52 inline bool oss_fill(std::basic_ostream<CharT, Traits>& os, SizeT size)
53 {
54  CharT c = os.fill();
55  CharT fill[] = { c, c, c, c, c, c, c, c };
56  enum {
57  chunk = sizeof fill / sizeof(CharT)
58  };
59  for ( ; size > chunk; size -= chunk) {
60  if (oss_put(os, fill, chunk) != chunk) {
61  return false;
62  }
63  }
64  return oss_put(os, fill, size) == size;
65 }

93 template<class CharT, class Traits, class SizeT>
94 inline std::basic_ostream<CharT, Traits>& ostream_string(std::basic_ostream<CharT, Traits>& os,
95  const CharT* data, SizeT size)
96 {
97  using stream = std::basic_ostream<CharT, Traits>;
98  oss_guard<CharT, Traits> guard(os);
99  typename stream::sentry entry(os);
100  if (entry) {
101  SizeT width = static_cast<SizeT>(os.width());
102  if (width <= size) {
103  if (detail::oss_put(os, data, size) != size) {
104  return os;
105  }
106  }
107  else if ((os.flags() & stream::adjustfield) == stream::left) {
108  if (detail::oss_put(os, data, size) != size ||
109  !detail::oss_fill(os, width - size)) {
110  return os;
111  }
112  }
113  else if (!detail::oss_fill(os, width - size) ||
114  detail::oss_put(os, data, size) != size) {
115  return os;
116  }
117  os.width(0);
118  }
119  guard.release();
120  return os;
121 }
