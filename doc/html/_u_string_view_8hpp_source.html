513  constexpr size_type find(UBasicStringView v, size_type pos = 0) const noexcept
514  {
515  if (pos > size() || v.empty() || v.size() > size() - pos) {
516  return npos;
517  }
518  const charT* cur = m_ptr + pos;
519  const charT* last = cend() - v.size() + 1;
520  for ( ; cur != last; ++cur) {
521  cur = traits::find(cur, last - cur, v[0]);
522  if (!cur) {
523  return npos;
524  }
525  if (traits::compare(cur, v.cbegin(), v.size()) == 0) {
526  return cur - m_ptr;
527  }
528  }
529  return npos;
530  }
531  constexpr size_type find(charT ch, size_type pos = 0) const noexcept
532  {
533  if (pos <= size()) {
534  const charT* ret_ptr = traits::find(m_ptr + pos, m_len - pos, ch);
535  if (ret_ptr) {
536  return ret_ptr - m_ptr;
537  }
538  }
539  return npos;
540  }

550  // rfind
551  constexpr size_type rfind(UBasicStringView s, size_type pos = npos) const noexcept
552  {
553  if (m_len < s.m_len) {
554  return npos;
555  }
556  if (pos > m_len - s.m_len) {
557  pos = m_len - s.m_len;
558  }
559  if (s.m_len == 0u) { // an empty string is always found
560  return pos;
561  }
562  for (const charT* cur = m_ptr + pos; ; --cur) {
563  if (traits::compare(cur, s.m_ptr, s.m_len) == 0) {
564  return cur - m_ptr;
565  }
566  if (cur == m_ptr) {
567  return npos;
568  }
569  }
570  }

605  // find_last_of
606  constexpr size_type find_last_of(UBasicStringView s, size_type pos = npos) const noexcept
607  {
608  if (s.len_ == 0u)
609  return npos;
610  if (pos >= len_)
611  pos = 0;
612  else
613  pos = len_ - (pos + 1);
614  const_reverse_iterator iter = std::find_first_of
615  (this->crbegin() + pos, this->crend(), s.cbegin(), s.cend(), traits::eq);
616  return iter == this->crend() ? npos : reverse_distance(this->crbegin(), iter);
617  }
