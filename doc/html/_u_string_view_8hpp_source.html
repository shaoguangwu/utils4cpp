556  constexpr size_type find(UBasicStringView v, size_type pos = 0) const noexcept
557  {
558  if (pos > size() || v.empty() || v.size() > size() - pos) {
559  return npos;
560  }
561  const CharT* cur = m_ptr + pos;
562  const CharT* last = cend() - v.size() + 1;
563  for ( ; cur != last; ++cur) {
564  cur = Traits::find(cur, last - cur, v[0]);
565  if (!cur) {
566  return npos;
567  }
568  if (Traits::compare(cur, v.cbegin(), v.size()) == 0) {
569  return cur - m_ptr;
570  }
571  }
572  return npos;
573  }
584  constexpr size_type find(CharT ch, size_type pos = 0) const noexcept
585  {
586  if (pos <= size()) {
587  const CharT* ret_ptr = Traits::find(m_ptr + pos, m_len - pos, ch);
588  if (ret_ptr) {
589  return ret_ptr - m_ptr;
590  }
591  }
592  return npos;
593  }

630  constexpr size_type rfind(UBasicStringView v, size_type pos = npos) const noexcept
631  {
632  if (m_len < v.m_len) {
633  return npos;
634  }
635  if (pos > m_len - v.m_len) {
636  pos = m_len - v.m_len;
637  }
638  if (v.m_len == 0u) { // an empty string is always found
639  return pos;
640  }
641  for (const CharT* cur = m_ptr + pos; ; --cur) {
642  if (Traits::compare(cur, v.m_ptr, v.m_len) == 0) {
643  return cur - m_ptr;
644  }
645  if (cur == m_ptr) {
646  return npos;
647  }
648  }
649  }

700  constexpr size_type find_first_of(UBasicStringView v, size_type pos = 0) const noexcept
701  {

706  this->cbegin() + pos, this->cend(), v.cbegin(), v.cend(), Traits::eq);

761  constexpr size_type find_last_of(UBasicStringView v, size_type pos = npos) const noexcept
762  {
763  if (v.m_len == 0u) {
764  return npos;
765  }
766  if (pos >= m_len) {
767  pos = 0;
768  } else {
769  pos = m_len - (pos + 1);
770  }

772  (this->crbegin() + pos, this->crend(), v.cbegin(), v.cend(), Traits::eq);
773  return iter == this->crend() ? npos : reverse_distance(this->crbegin(), iter);
774  }

832  constexpr size_type find_first_not_of(UBasicStringView v, size_type pos = 0) const noexcept
833  {

897  constexpr size_type find_last_not_of(UBasicStringView v, size_type pos = npos) const noexcept
898  {
899  if (pos >= m_len) {
900  pos = m_len - 1;
901  }
902  if (v.m_len == 0u) {
903  return pos;

905  pos = m_len - (pos + 1);

907  return iter == this->crend() ? npos : reverse_distance(this->crbegin(), iter);
908  }

952  return find_last_not_of(UBasicStringView(s), pos);
953  }
954 private:
955  template <typename r_iter>
956  size_type reverse_distance(r_iter first, r_iter last) const noexcept
957  {
958  // Portability note here: std::distance is not NOEXCEPT, but calling it with a string_view::reverse_iterator will not throw.
959  return m_len - 1 - std::distance(first, last);
960  }

962  template <typename Iterator>
963  Iterator find_not_of(Iterator first, Iterator last, UBasicStringView v) const noexcept
964  {
965  for (; first != last; ++first) {
966  if (0 == Traits::find(v.ptr_, v.m_len, *first)) {
967  return first;
968  }
969  }
970  return last;
971  }
