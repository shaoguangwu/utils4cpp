128  static int compare(const StringT& str1, UBasicStringView<char_type> str2,
129  UCaseSensitivity cs = UCaseSensitive, const std::locale& loc = std::locale())
130  {
131  size_type lhs_sz = str1.size();
132  size_type rhs_sz = str2.size();
133  if (cs == UCaseSensitive) {
134  if (int result = char_traits_type::compare(str1.data(), str2.data(), std::min(lhs_sz, rhs_sz))) {
135  return result;
136  }
137  }
138  else {
139  if (int result = _strucmp(str1.data(), str2.data(), std::min(lhs_sz, rhs_sz))) {
140  return result;
141  }
142  }
143 
144  if (lhs_sz < rhs_sz) {
145  return -1;
146  }
147  if (lhs_sz > rhs_sz) {
148  return 1;
149  }
150 
151  return 0;
152  }
