69 static std::int64_t julianDayFromDate(int year, int month, int day)
70 {
71  // Adjust for no year 0
72  if (year < 0)
73  ++year;
74 
75  /*
76  * Math from The Calendar FAQ at http://www.tondering.dk/claus/cal/julperiod.php
77  * This formula is correct for all julian days, when using mathematical integer
78  * division (round to negative infinity), not c++11 integer division (round to zero)
79  */
80  int a = floordiv(14 - month, 12);
81  std::int64_t y = (std::int64_t)year + 4800 - a;
82  int m = month + 12 * a - 3;
83  return day + floordiv(153 * m + 2, 5) + 365 * y + floordiv(y, 4) - floordiv(y, 100) + floordiv(y, 400) - 32045;
84 }

86 static void getDateFromJulianDay(std::int64_t julianDay, int& year, int& month, int& day)
87 {
88  /*
89  * Math from The Calendar FAQ at http://www.tondering.dk/claus/cal/julperiod.php
90  * This formula is correct for all julian days, when using mathematical integer
91  * division (round to negative infinity), not c++11 integer division (round to zero)
92  */
93  std::int64_t a = julianDay + 32044;
94  std::int64_t b = floordiv(4 * a + 3, 146097);
95  std::int64_t c = a - floordiv(146097 * b, 4);
96 
97  std::int64_t d = floordiv(4 * c + 3, 1461);
98  std::int64_t e = c - floordiv(1461 * d, 4);
99  std::int64_t m = floordiv(5 * e + 2, 153);
100 
101  day = static_cast<int>(e - floordiv(153 * m + 2, 5) + 1);
102  month = static_cast<int>(m + 3 - 12 * floordiv(m, 10));
103  year = static_cast<int>(100 * b + d - 4800 + floordiv(m, 10));
104 
105  // Adjust for no year 0
106  if (year <= 0)
107  --year;
108 }

505 UDate UDate::addMonths(int months) const
506 {
507  if (!isValid())
508  return UDate();
509  if (!months)
510  return *this;
511 
512  int old_y = m_dt.year;
513  int y = m_dt.year;
514  int m = m_dt.mon;
515  int d = m_dt.mday;
516 
517  bool increasing = months > 0;
518 
519  while (months != 0) {
520  if (months < 0 && months + 12 <= 0) {
521  y--;
522  months += 12;
523  }
524  else if (months < 0) {
525  m += months;
526  months = 0;
527  if (m <= 0) {
528  --y;
529  m += 12;
530  }
531  }
532  else if (months - 12 >= 0) {
533  y++;
534  months -= 12;
535  }
536  else if (m == 12) {
537  y++;
538  m = 0;
539  }
540  else {
541  m += months;
542  months = 0;
543  if (m > 12) {
544  ++y;
545  m -= 12;
546  }
547  }
548  }
549 
550  // was there a sign change?
551  if ((old_y > 0 && y <= 0) ||
552  (old_y < 0 && y >= 0))
553  // yes, adjust the date by +1 or -1 years
554  y += increasing ? +1 : -1;
555 
556  return fixedDate(y, m, d);
557 }

764 int UDate::dayOfYear(int year, int month, int day)
765 {
766  if (!isValid(year, month, day))
767  return 0;
768 
769  int days = 0;
770  for (int i = 1; i < month; ++i)
771  days += monthDays[i];
772 
773  if (isLeapYear(year) && month > 2)
774  return days + day + 1;
775  else
776  return days + day;
777 }

793 int UDate::dayOfWeek(int year, int month, int day)
794 {
795  int weekday = 0;
796  if (!isValid(year, month, day))
797  return weekday;
798 
799  if (beforGregorianCalendar(year, month, day))
800  weekday = (day + 1 + 2 * month + 3 * (month + 1) / 5 + year + year / 4 + 5) % 7;
801  else
802  weekday = (day + 2 * month + 3 * (month + 1) / 5 + year + year / 4 - year / 100 + year / 400 + 1) % 7;
803 
804  if (weekday == 0)
805  weekday = 7;
806 
807  return weekday;
808 }
