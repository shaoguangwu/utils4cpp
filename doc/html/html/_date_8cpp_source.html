68 static std::int64_t julianDayFromDate(int year, int month, int day)
69 {
70  // Adjust for no year 0
71  if (year < 0)
72  ++year;
73 
74  /*
75  * Math from The Calendar FAQ at http://www.tondering.dk/claus/cal/julperiod.php
76  * This formula is correct for all julian days, when using mathematical integer
77  * division (round to negative infinity), not c++11 integer division (round to zero)
78  */
79  int a = floordiv(14 - month, 12);
80  std::int64_t y = (std::int64_t)year + 4800 - a;
81  int m = month + 12 * a - 3;
82  return day + floordiv(153 * m + 2, 5) + 365 * y + floordiv(y, 4) - floordiv(y, 100) + floordiv(y, 400) - 32045;
83 }

85 static void getDateFromJulianDay(std::int64_t julianDay, int& year, int& month, int& day)
86 {
87  /*
88  * Math from The Calendar FAQ at http://www.tondering.dk/claus/cal/julperiod.php
89  * This formula is correct for all julian days, when using mathematical integer
90  * division (round to negative infinity), not c++11 integer division (round to zero)
91  */
92  std::int64_t a = julianDay + 32044;
93  std::int64_t b = floordiv(4 * a + 3, 146097);
94  std::int64_t c = a - floordiv(146097 * b, 4);
95 
96  std::int64_t d = floordiv(4 * c + 3, 1461);
97  std::int64_t e = c - floordiv(1461 * d, 4);
98  std::int64_t m = floordiv(5 * e + 2, 153);
99 
100  day = static_cast<int>(e - floordiv(153 * m + 2, 5) + 1);
101  month = static_cast<int>(m + 3 - 12 * floordiv(m, 10));
102  year = static_cast<int>(100 * b + d - 4800 + floordiv(m, 10));
103 
104  // Adjust for no year 0
105  if (year <= 0)
106  --year;
107 }

480 Date Date::addMonths(int months) const
481 {
482  if (!isValid())
483  return Date();
484  if (!months)
485  return *this;
486 
487  int old_y = m_dt.year;
488  int y = m_dt.year;
489  int m = m_dt.mon;
490  int d = m_dt.mday;
491 
492  bool increasing = months > 0;
493 
494  while (months != 0) {
495  if (months < 0 && months + 12 <= 0) {
496  y--;
497  months += 12;
498  }
499  else if (months < 0) {
500  m += months;
501  months = 0;
502  if (m <= 0) {
503  --y;
504  m += 12;
505  }
506  }
507  else if (months - 12 >= 0) {
508  y++;
509  months -= 12;
510  }
511  else if (m == 12) {
512  y++;
513  m = 0;
514  }
515  else {
516  m += months;
517  months = 0;
518  if (m > 12) {
519  ++y;
520  m -= 12;
521  }
522  }
523  }
524 
525  // was there a sign change?
526  if ((old_y > 0 && y <= 0) ||
527  (old_y < 0 && y >= 0))
528  // yes, adjust the date by +1 or -1 years
529  y += increasing ? +1 : -1;
530 
531  return fixedDate(y, m, d);
532 }

751 int Date::dayOfWeek(int year, int month, int day)
752 {
753  int weekday = 0;
754  if (!isValid(year, month, day))
755  return weekday;
756 
757  if (beforGregorianCalendar(year, month, day))
758  weekday = (day + 1 + 2 * month + 3 * (month + 1) / 5 + year + year / 4 + 5) % 7;
759  else
760  weekday = (day + 2 * month + 3 * (month + 1) / 5 + year + year / 4 - year / 100 + year / 400 + 1) % 7;
761 
762  if (weekday == 0)
763  weekday = 7;
764 
765  return weekday;
766 }
