63 static std::int64_t julianDayFromDate(int year, int month, int day)
64 {
65  // Adjust for no year 0
66  if (year < 0)
67  ++year;
68 
69  /*
70  * Math from The Calendar FAQ at http://www.tondering.dk/claus/cal/julperiod.php
71  * This formula is correct for all julian days, when using mathematical integer
72  * division (round to negative infinity), not c++11 integer division (round to zero)
73  */
74  int a = floordiv(14 - month, 12);
75  std::int64_t y = (std::int64_t)year + 4800 - a;
76  int m = month + 12 * a - 3;
77  return day + floordiv(153 * m + 2, 5) + 365 * y + floordiv(y, 4) - floordiv(y, 100) + floordiv(y, 400) - 32045;
78 }

80 static void getDateFromJulianDay(std::int64_t julianDay, int& year, int& month, int& day)
81 {
82  /*
83  * Math from The Calendar FAQ at http://www.tondering.dk/claus/cal/julperiod.php
84  * This formula is correct for all julian days, when using mathematical integer
85  * division (round to negative infinity), not c++11 integer division (round to zero)
86  */
87  std::int64_t a = julianDay + 32044;
88  std::int64_t b = floordiv(4 * a + 3, 146097);
89  std::int64_t c = a - floordiv(146097 * b, 4);
90 
91  std::int64_t d = floordiv(4 * c + 3, 1461);
92  std::int64_t e = c - floordiv(1461 * d, 4);
93  std::int64_t m = floordiv(5 * e + 2, 153);
94 
95  day = static_cast<int>(e - floordiv(153 * m + 2, 5) + 1);
96  month = static_cast<int>(m + 3 - 12 * floordiv(m, 10));
97  year = static_cast<int>(100 * b + d - 4800 + floordiv(m, 10));
98 
99  // Adjust for no year 0
100  if (year <= 0)
101  --year;
102 }

425 Date Date::addMonths(int months) const
426 {
427  if (!isValid())
428  return Date();
429  if (!months)
430  return *this;
431 
432  int old_y = m_dt.year;
433  int y = m_dt.year;
434  int m = m_dt.mon;
435  int d = m_dt.mday;
436 
437  bool increasing = months > 0;
438 
439  while (months != 0) {
440  if (months < 0 && months + 12 <= 0) {
441  y--;
442  months += 12;
443  }
444  else if (months < 0) {
445  m += months;
446  months = 0;
447  if (m <= 0) {
448  --y;
449  m += 12;
450  }
451  }
452  else if (months - 12 >= 0) {
453  y++;
454  months -= 12;
455  }
456  else if (m == 12) {
457  y++;
458  m = 0;
459  }
460  else {
461  m += months;
462  months = 0;
463  if (m > 12) {
464  ++y;
465  m -= 12;
466  }
467  }
468  }
469 
470  // was there a sign change?
471  if ((old_y > 0 && y <= 0) ||
472  (old_y < 0 && y >= 0))
473  // yes, adjust the date by +1 or -1 years
474  y += increasing ? +1 : -1;
475 
476  return fixedDate(y, m, d);
477 }

690 int Date::dayOfWeek(int year, int month, int day)
691 {
692  int weekday = 0;
693  if (!isValid(year, month, day))
694  return weekday;
695 
696  if (beforGregorianCalendar(year, month, day))
697  weekday = (day + 1 + 2 * month + 3 * (month + 1) / 5 + year + year / 4 + 5) % 7;
698  else
699  weekday = (day + 2 * month + 3 * (month + 1) / 5 + year + year / 4 - year / 100 + year / 400 + 1) % 7;
700 
701  if (weekday == 0)
702  weekday = 7;
703 
704  return weekday;
705 }
