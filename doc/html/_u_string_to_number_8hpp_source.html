70  try {
71  if constexpr (std::is_signed_v<primitive_t<NumberT>>) {
72  if constexpr (sizeof(NumberT) <= sizeof(int)) {
73  return static_cast<primitive_t<NumberT>>(std::stoi(str, pos, base));
74  }

76  return static_cast<primitive_t<NumberT>>(std::stol(str, pos, base));
