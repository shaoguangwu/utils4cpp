65 template<class StringT>
66 void removeCharacterSelf(StringT& str, typename StringT::value_type ch, int option = 0)
67 {
68  using CharT = typename StringT::value_type;
69  using SizeT = typename StringT::size_type;
70 
71  if (str.empty()) {
72  return;
73  }
74 
75  if (option != 0) {
76  auto pos = option < 0 ? str.find(ch) : str.rfind(ch);
77  if (pos != StringT::npos) {
78  str.erase(pos, 1);
79  return;
80  }
81  }
82  else { // option == 0
83  auto pos = str.find(ch);
84  if (pos != StringT::npos) {
85  std::unique_ptr<CharT[]> buff(new CharT[str.size() + 1]);
86  SizeT len = 0;
87  SizeT pos1 = 0;
88  while (pos != StringT::npos) {
89  std::copy_n(str.begin() + pos1, pos - pos1, buff.get() + len);
90  len += pos - pos1;
91  pos1 = ++pos;
92  pos = str.find(ch, pos1);
93  }
94  std::copy_n(str.begin() + pos1, str.size() - pos1, buff.get() + len);
95  len += str.size() - pos1;
96  str = StringT(buff.get(), len);
97  }
98  }
99 }

187 template<class StringT>
188 bool containsCharacter(const StringT& str, typename StringT::value_type ch,
189  UCaseSensitivity cs = UCaseSensitive, const std::locale& loc = std::locale())
190 {
191  if (!str.empty()) {
192  if (cs == UCaseSensitive) {
193  return str.find(ch) != StringT::npos;
194  }
195  else {
196  using CharT = typename StringT::value_type;
197  CharT chl = std::tolower(ch, loc);
198  CharT chu = std::toupper(ch, loc);
199  for (const auto& c : str) {
200  if (c == chl || c == chu) {
201  return true;
202  }
203  }
204  }
205  }
206  return false;
207 }

263 template<class StringT>
264 UStringVector<StringT> splitString(const StringT& str, typename StringT::value_type delim)
265 {
266  using SizeT = typename StringT::size_type;
267  UStringVector<StringT> result;
268  if (!str.empty()) {
269  SizeT beg = 0;
270  SizeT pos = str.find(delim);
271  while (pos != StringT::npos) {
272  result.emplace_back(str.substr(beg, pos - beg));
273  beg = ++pos;
274  pos = str.find(delim, beg);
275  }
276  if (beg > 0) {
277  result.emplace_back(str.substr(beg));
278  }
279  }
280 
281  return result;
282 }

531 template<>
532 std::string convertString(const std::wstring& str, const char* locale)
533 {
534  std::string ctype_locale;
535  auto prev_ctype_loc = std::setlocale(LC_CTYPE, locale);
536  if (prev_ctype_loc) {
537  ctype_locale = prev_ctype_loc;
538  }
539 
540  std::string result;
541  std::mbstate_t state = std::mbstate_t();
542  const wchar_t* wstr = str.c_str();
543 
544 #ifdef _MSC_VER
545 # pragma warning(push)
546 # pragma warning(disable:4996)
547 #endif // _MSC_VER
548  auto len = std::wcsrtombs(nullptr, &wstr, 0, &state) + 1;
549  if (len > 1) {
550  std::unique_ptr<char[]> buf(new char[len]);
551  std::wcsrtombs(buf.get(), &wstr, len, &state);
552  result = std::string(buf.get(), len - 1);
553  }
554 
555 #ifdef _MSC_VER
556 # pragma warning(pop)
557 #endif // _MSC_VER
558 
559  std::setlocale(LC_CTYPE, ctype_locale.c_str()); // restore locale-ctype
560  return result;
561 }
