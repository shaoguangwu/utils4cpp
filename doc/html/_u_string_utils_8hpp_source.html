183 template<class StringT>
184 void removeCharacterSelf(StringT& str, typename StringT::value_type ch, int option)
185 {
186  using CharT = typename StringT::value_type;
187  using SizeT = typename StringT::size_type;
188 
189  if (str.empty()) {
190  return;
191  }
192 
193  if (option != 0) {
194  auto pos = option < 0 ? str.find(ch) : str.rfind(ch);
195  if (pos != StringT::npos) {
196  str.erase(pos, 1);
197  return;
198  }
199  }
200  else { // option == 0
201  auto pos = str.find(ch);
202  if (pos != StringT::npos) {
203  std::unique_ptr<CharT[]> buff(new CharT[str.size() + 1]);
204  SizeT len = 0;
205  SizeT pos1 = 0;
206  while (pos != StringT::npos) {
207  std::copy_n(str.begin() + pos1, pos - pos1, buff.get() + len);
208  len += pos - pos1;
209  pos1 = ++pos;
210  pos = str.find(ch, pos1);
211  }
212  std::copy_n(str.begin() + pos1, str.size() - pos1, buff.get() + len);
213  len += str.size() - pos1;
214  str = StringT(buff.get(), len);
215  }
216  }
217 }

305 template<class StringT>
306 bool containsCharacter(const StringT& str, typename StringT::value_type ch, CaseSensitivity cs, const std::locale& loc)
307 {
308  if (!str.empty()) {
309  if (cs == CaseSensitive) {
310  return str.find(ch) != StringT::npos;
311  }
312  else {
313  using CharT = typename StringT::value_type;
314  CharT chl = std::tolower(ch, loc);
315  CharT chu = std::toupper(ch, loc);
316  for (const auto& c : str) {
317  if (c == chl || c == chu) {
318  return true;
319  }
320  }
321  }
322  }
323  return false;
324 }

381 {

383  UStringVector<StringT> result;
384  if (!str.empty()) {
385  SizeT beg = 0;
386  SizeT pos = str.find(delim);

388  result.emplace_back(str.substr(beg, pos - beg));
389  beg = ++pos;
390  pos = str.find(delim, beg);
391  }
392  if (beg > 0) {
393  result.emplace_back(str.substr(beg));
394  }
395  }
396 
397  return result;

647 template<>
648 std::string convertString(const std::wstring& str, const char* locale)
649 {
650  std::string ctype_locale;
651  auto prev_ctype_loc = std::setlocale(LC_CTYPE, locale);
652  if (prev_ctype_loc) {
653  ctype_locale = prev_ctype_loc;
654  }
655 
656  std::string result;
657  std::mbstate_t state = std::mbstate_t();
658  const wchar_t* wstr = str.c_str();
659 
660 #ifdef _MSC_VER
661 # pragma warning(push)
662 # pragma warning(disable:4996)
663 #endif // _MSC_VER
664  auto len = std::wcsrtombs(nullptr, &wstr, 0, &state) + 1;
665  if (len > 1) {
666  std::unique_ptr<char[]> buf(new char[len]);
667  std::wcsrtombs(buf.get(), &wstr, len, &state);
668  result = std::string(buf.get(), len - 1);
669  }
670 
671 #ifdef _MSC_VER
672 # pragma warning(pop)
673 #endif // _MSC_VER
674 
675  std::setlocale(LC_CTYPE, ctype_locale.c_str()); // restore locale-ctype
676  return result;
677 }
