59 template<class InputT, class OutputT,
60  class = if_char<InputT>, class = if_character<OutputT>>
61 class UCharCvt
62 {
63 public:
64  using input_type = primitive_t<InputT>;
65  using output_type = primitive_t<OutputT>;
66 
67  static std::optional<output_type> convert(input_type c)
68  {
69  if constexpr (std::is_same_v<c, wchar_t>) {
70  return charToWChar(c);
71  }
72  else if constexpr (std::is_same_v<c, char16_t>) {
73  return charToU16Char(c);
74  }
75  else if constexpr (std::is_same_v<c, char32_t>) {
76  return charToU32Char(c);
77  }
78 #if UTILS4CPP_HAS_CHAR8T
79  else if constexpr (std::is_same_v<c, char8_t>) {
80  return charToU8Char(c);
81  }
82 #endif // UTILS4CPP_HAS_CHAR8T
83  else {
84  return std::nullopt;
85  }
86  }
87 };
