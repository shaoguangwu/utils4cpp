382  size_type removeDuplicates()
383  {
384  auto& self = *this;
385  size_type n = self.size();
386  size_type j = 0;
387  std::set<StringT> seen;
388  size_type setSize = 0;
389  for (size_type i = 0; i < n; ++i) {
390  const StringT& str = self[i];
391  seen.insert(str);
392  if (setSize == seen.size()) {
393  continue;
394  }
395  ++setSize;
396  if (j != i) {
397  self.swapItems(i, j);
398  }
399  ++j;
400  }
401  if (n != j) {
402  self.erase(self.iteratorAt(j), self.end());
403  }
404  return n - j;
405  }
