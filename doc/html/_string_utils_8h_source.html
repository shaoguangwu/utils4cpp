196 template<typename ... Args>
197 std::string formatString(const std::string& format, Args ... args)
198 {
199  auto size = std::snprintf(nullptr, 0, format.c_str(), args ...) + 1; // Extra space for '\0'
200  std::unique_ptr<char[]> buf(new char[size]);
201  std::snprintf(buf.get(), size, format.c_str(), args ...);
202  return { buf.get(), buf.get() + size - 1 }; // We don't want the '\0' inside
203 }

211 template<typename ... Args>
212 std::wstring formatString(const std::wstring& format, Args ... args)
213 {
214  auto size = std::swprintf(nullptr, 0, format.c_str(), args ...) + 1; // Extra space for '\0'
215  std::unique_ptr<wchar_t[]> buf(new wchar_t[size]);
216  std::swprintf(buf.get(), size, format.c_str(), args ...);
217  return { buf.get(), buf.get() + size - 1 }; // We don't want the '\0' inside
218 }
